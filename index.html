// ฟังก์ชันดาวน์โหลดและแคปภาพ (ฉบับปรับปรุงเพื่อความเสถียรบน GitHub Pages) async function captureAndDownload(format, filename) { const capArea = document.getElementById('captureArea'); // 1. เตรียมพื้นที่แคปภาพให้พร้อมเรนเดอร์ (ดึงมาไว้ในจุดที่มองเห็นแต่ไม่รบกวนผู้ใช้) capArea.style.left = '0px'; capArea.style.top = '0px'; capArea.style.zIndex = '9000'; capArea.style.position = 'fixed'; capArea.style.background = 'white'; window.scrollTo(0, 0); // 2. เพิ่มเวลารอเพื่อให้แน่ใจว่า Font และรูปภาพลายเซ็น (Base64) แสดงผลครบ 100% // ปรับจาก 800ms เป็น 1500ms เพื่อความชัวร์บนมือถือ await new Promise(res => setTimeout(res, 1500)); try { if (format === 'jpg' || format === 'both') { const canvas = await html2canvas(capArea, { scale: 2, useCORS: true, logging: false, scrollY: 0, windowWidth: 800 }); const imgData = canvas.toDataURL("image/jpeg", 0.95); const dlLink = document.createElement("a"); dlLink.href = imgData; dlLink.download = filename + '.jpg'; document.body.appendChild(dlLink); dlLink.click(); document.body.removeChild(dlLink); } if (format === 'pdf' || format === 'both') { // ตั้งค่า PDF ให้แม่นยำขึ้น const opt = { margin: [0.3, 0.3, 0.3, 0.3], // [top, left, bottom, right] filename: filename + '.pdf', image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, useCORS: true, letterRendering: true, scrollY: 0, windowWidth: 800 }, jsPDF: { unit: 'in', format: 'a4', orientation: 'portrait' }, pagebreak: { mode: ['avoid-all', 'css', 'legacy'] } }; // เรียกใช้ html2pdf แบบ Promise เพื่อดักจับ Error await html2pdf().set(opt).from(capArea).save(); } } catch (error) { console.error("Export Error:", error); throw new Error("เกิดข้อผิดพลาดในการสร้างไฟล์: " + error.message); } finally { // 3. ซ่อนพื้นที่แคปภาพกลับที่เดิม capArea.style.left = '-9999px'; capArea.style.position = 'absolute'; } }
